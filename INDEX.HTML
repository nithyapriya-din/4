let obj1 = { name: "person 1", age: 5 }; let obj2 = { age: 5, name: "person 1" }; // Step 1: Convert objects to strings let str1 = JSON.stringify(obj1); let str2 = JSON.stringify(obj2); // Step 2: Parse strings back to objects let parsedObj1 = JSON.parse(str1); let parsedObj2 = JSON.parse(str2); // Step 3: Compare objects const areEqual = deepEqual(parsedObj1, parsedObj2); console.log(areEqual); // Function to deeply compare two objects function deepEqual(obj1, obj2) { if (obj1 === obj2) return true; if (typeof obj1 !== "object" || typeof obj2 !== "object") return false; const keys1 = Object.keys(obj1); const keys2 = Object.keys(obj2); if (keys1.length !== keys2.length) return false; for (let key of keys1) { if (!keys2.includes(key) || !deepEqual(obj1[key], obj2[key])) { return false; } } return true; }
